239template<typename T>
240void frequency_list<T>::swap_nodes(std::shared_ptr<node> a, std::shared_ptr<node> b) {
241 if(a == nullptr || b == nullptr || a->next != b)
242 return;
243
244 // Attach b's next to a and a's prev to b
245 a->next = b->next;
246 b->prev = a->prev;
247
248 // If b's next exists, attach a to its prev
249 if(b->next)
250 b->next->prev = a;
251
252 // If a's prev exists(i.e., a is not the head), attach b to its next
253 if(a->prev.lock())
254 a->prev.lock()->next = b;
255
256 // Now, attach a to b's next and b to a's prev
257 b->next = a;
258 a->prev = b;
259
260 // If the node a was the head node, update the head to b
261 if (head == a)
262 head = b;
263}

265template<typename T>
266void frequency_list<T>::update_list_sequence() {
267 auto current_node = head;
268
269 while (current_node != nullptr && current_node->next != nullptr) {
270 if (current_node->freq < current_node->next->freq) {
271 swap_nodes(current_node, current_node->next);
272
273 if(current_node->prev.lock())
274 current_node = current_node->prev.lock();
275
276 } else
277 current_node = current_node->next;
278 }

293void frequency_list<T>::push_front(T data) {
294 auto Iterator = head;
295
296 // Iterating over the list to find if the same data is already present
297 while (Iterator != nullptr) {
298 if (Iterator->data == data) {
299 // if data is found, increase frequency and re-sort the list
300 Iterator->freq++;
301 update_list_sequence();
302 return;
303 }
304 Iterator = Iterator->next;
305 }
306
307 // if data is not found in the list, add new node at front
308 auto new_node = std::make_shared<node>(data, 1, std::weak_ptr<node>(), head);
309
310 if (head != nullptr) {
311 head->prev = new_node;
312 }
313
314 head = new_node;
315
316 size++;
317}

319template<typename T>
320void frequency_list<T>::erase(T key) {
321 auto iterator = head;
322
323 // special case: deleting the head node
324 if (head != nullptr && head->data == key) {
325 head = head->next;
326 if(head)
327 head->prev.reset();
328 size--;
329 return;
330 }
331
332 // iterate over the rest nodes
333 while (iterator != nullptr) {
334 if (iterator->data == key) {
335 // connect previous node and next node
336 if(auto sharedPrev = iterator->prev.lock()){
337 sharedPrev->next = iterator->next;
338 }
339 if (iterator->next) {
340 iterator->next->prev = iterator->prev;
341 }
342 // deleting the node by falling out of scope
343 size--;
344 return;
345 }
346 iterator = iterator->next;
347 }
