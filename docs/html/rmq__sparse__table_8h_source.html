36 void build(const std::vector<T> &_values) {

38 n = int(values.size());
39 int levels = highest_bit(n) + 1;
40 range_low.resize(levels);
41
42 for (int k = 0; k < levels; k++)
43 range_low[k].resize(n - (1 << k) + 1);
44
45 for (int i = 0; i < n; i++)
46 range_low[0][i] = i;
47
48 for (int k = 1; k < levels; k++)
49 for (int i = 0; i <= n - (1 << k); i++)
50 range_low[k][i] = better_index(range_low[k - 1][i], range_low[k - 1][i + (1 << (k - 1))]);
51 }
52
53 // Note: breaks ties by choosing the largest index.
54 int query_index(int a, int b) const {
55 assert(0 <= a && a < b && b <= n);
56 int level = highest_bit(b - a);
57 return better_index(range_low[level][a], range_low[level][b - (1 << level)]);
58 }
