37namespace graph_visualization {
38/*
39 *visualize function.
40 *@param __generate: string that contains the needed data to generate the .dot
41 *file.
42 *@param newFileName: if needed, the filename can be changed(default
43 *"unnamed.dot")
44 */
45void visualize(std::string &__generate,
46 std::string newFileName = "unnamed.dot") {
47 auto start_time = std::chrono::high_resolution_clock::now();
48 try {
49 if (newFileName.size() < 5 ||
50 newFileName.substr(newFileName.length() - 4) != ".dot")
51 newFileName += ".dot";
52 // newFileName = "examples/" + newFileName;
53 // Open the file for writing
54 std::ofstream outFile(newFileName);
55
56 // Check if the file is successfully opened
57 if (outFile.is_open()) {
58 // Write the DOT format header
59 outFile << "graph G{" << std::endl;
60
61 // Generate DOT code recursively
62 outFile << __generate;
63 // Write the DOT format footer
64 outFile << "}" << std::endl;
65
66 // Close the file
67 outFile.close();
68
69 std::cout << green << "Visualization file '" << newFileName
70 << "' created successfully." << std::endl
71 << reset;
72 auto end_time = std::chrono::high_resolution_clock::now();
73 auto duration = std::chrono::duration_cast<std::chrono::microseconds>(
74 end_time - start_time);
75 double runtime_sec = static_cast<double>(duration.count()) / 1e6;
76
77 std::cout << blue << "Visualization runtime: " << runtime_sec << " sec"
78 << std::endl
79 << reset;
80 std::string openCommand = OPEN_COMMAND + std::string(" ") + newFileName;
81 system(openCommand.c_str());
82 } else {
83 std::cerr << red << "Error: Unable to open file '" << newFileName
84 << "' for writing." << std::endl
85 << reset;
86 }
87 } catch (const std::exception &e) {
88 std::cerr << red << "Error: " << e.what() << std::endl << reset;
89 }
90};
