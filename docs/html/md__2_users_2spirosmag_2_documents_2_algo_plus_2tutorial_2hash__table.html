retrieve:

#include <hash_table.h>
hash_table<std::string, int> ht;
ht.insert("apple", 1);
ht.insert("banana", 2);
ht.insert("carrot", 3);
// retrieves the value associated with the given key.
auto opt_value = ht.retrieve("banana");
if(opt_value) {
std::cout << *opt_value << '\n'; // will print 2
} else {
std::cout << "Element not found" << '\n';
