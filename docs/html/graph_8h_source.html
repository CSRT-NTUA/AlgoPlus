808template <typename T> int64_t weighted_graph<T>::shortest_path(T start, T end) {
809 if (_elements.find(start) == _elements.end()) {
810 std::cout << "Element: " << start << " is not found in the Graph" << '\n';
811 return -1;
812 }
813 if (_elements.find(end) == _elements.end()) {
814 std::cout << "Element: " << end << " is not found in the Graph" << '\n';
815 return -1;
816 }
817
818 if (!cycle() && _type == "directed") {
819 std::vector<T> top_sort = topological_sort();
820 std::reverse(top_sort.begin(), top_sort.end());
821 std::stack<T> s;
822 std::unordered_map<T, int64_t> dist;
823 for (auto &x : _elements) {
824 dist[x] = INT_MAX;
825 }
826 dist[start] = 0;
827 while (!top_sort.empty()) {
828 auto current = top_sort.back();
829 top_sort.pop_back();
830 if (dist[current] != INT_MAX) {
831 for (std::pair<T, int64_t> &x : adj[current]) {
832 if (dist[x.first] > dist[current] + x.second) {
833 dist[x.first] = dist[current] + x.second;
834 top_sort.push_back(x.first);
835 }
836 }
837 }
838 }
839 return (dist[end] != INT_MAX) ? dist[end] : -1;
840 } else {
841 std::unordered_map<T, int64_t> dist;
842 for (auto &x : _elements) {
843 dist[x] = INT_MAX;
844 }
845 std::priority_queue<std::pair<int64_t, T>,
846 std::vector<std::pair<int64_t, T>>,
847 std::greater<std::pair<int64_t, T>>>
848 pq;
849 pq.push(std::make_pair(0, start));
850 dist[start] = 0;
851 while (!pq.empty()) {
852 T currentNode = pq.top().second;
853 T currentDist = pq.top().first;
854 pq.pop();
855 for (std::pair<T, int64_t> &edge : adj[currentNode]) {
856 if (currentDist + edge.second < dist[edge.first]) {
857 dist[edge.first] = currentDist + edge.second;
858 pq.push(std::make_pair(dist[edge.first], edge.first));
859 }
860 }
861 }
862 return (dist[end] != INT_MAX) ? dist[end] : -1;
863 }
864 return -1;

1041

1043 if (color.find(x) == color.end()) {
1044 q.push({x, 0});

1048 q.pop();

1135template <typename T>
1136std::unordered_map<T, double> weighted_graph<T>::bellman_ford(T start) {
1137 std::unordered_map<T, double> dist;
1138 // Initialize the distance to all nodes to be infinity
1139 // except for the starting node which is zero.
1140 for (auto it : adj)
1141 dist[it.first] = std::numeric_limits<double>::infinity();
1142 dist[start] = 0;
1143
1144 // Get number of vertices present in the graph
1145 int v = adj.size();
1146
1147 // For each vertex, apply relaxation for all the edges
1148 for (int i = 0; i < v - 1; i++)
1149 for (const auto j : adj)
1150 for (const auto edge : adj[j.first])
1151 if (dist[j.first] + edge.second < dist[edge.first])
1152 dist[edge.first] = dist[j.first] + edge.second;
1153
1154 // Run algorithm a second time to detect which nodes are part
1155 // of a negative cycle. A negative cycle has occurred if we
1156 // can find a better path beyond the optimal solution.
1157 for (int i = 0; i < v - 1; i++)
1158 for (const auto j : adj)
1159 for (const auto edge : adj[j.first])
1160 if (dist[j.first] + edge.second < dist[edge.first])
1161 dist[edge.first] = -std::numeric_limits<double>::infinity();
1162
1163 // Return the array containing the shortest distance to every node
1164 return dist;
1165}
