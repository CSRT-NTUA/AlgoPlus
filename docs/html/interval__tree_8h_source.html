302 std::shared_ptr<node> _remove(std::shared_ptr<node> root, interval i) {
303 if (!root) {
304 return nullptr;
305 }
306 std::shared_ptr<node> p = root;
307 while (p) {
308 if (p->i->low > i.low) {
309 p = p->left;
310 } else if (p->i->low < i.low) {
311 p = p->right;
312 } else {
313 if (!p->right && !p->left) {
314 return nullptr;
315 } else if (p->right && !p->left) {
316 std::shared_ptr<node> temp = root->right;
317 return temp;
318 } else if (p->left && !p->right) {
319 std::shared_ptr<node> temp = p->left;
320 return temp;
321 } else {
322 std::shared_ptr<node> temp = root;
323 while (temp && temp->left) {
324 temp = temp->left;
325 }
326 p->i = temp->i;
327 p->max = temp->max;
328 p->right = _remove(p->right, *temp->i);
