181 q.push(root);

243 std::shared_ptr<node> splay(std::shared_ptr<node> _node, T key) {
244 if (!_node || _node->info == key) {
245 return _node;
246 }
247 if (_node->info > key) {
248 if (_node->left == NULL)
249 return _node;
250 if (_node->left->info > key) {
251 _node->left->left = splay(_node->left->left, key);
252 _node = rrotate(_node);
253 } else if (_node->left->info < key) {
254 _node->left->right = splay(_node->left->right, key);
255 if (_node->left->right != NULL)
256 _node->left = lrotate(_node->left);
257 }
258 return (_node->left == NULL) ? _node : rrotate(_node);
259 } else {
260 if (_node->right == NULL)
261 return _node;
262 if (_node->right->info > key) {
263 _node->right->left = splay(_node->right->left, key);
264 if (_node->right->left != NULL)
265 _node->right = rrotate(_node->right);
266 } else if (_node->right->info < key) {
267 _node->right->right = splay(_node->right->right, key);
268 _node = lrotate(_node);
269 }
